title Turno de jugador

participant usuario
participant personajeJugador
participant batalla
participant personajeEnemigo

autoactivation on

batalla -> usuario: iniciarTurno()
usuario -> usuario: seleccionarAccion()

alt accion = ataqueMagico
    usuario -> personajeJugador: usarMagia()
    personajeJugador -> personajeJugador: manaDisponible()

    alt manaDisponible == true
        personajeJugador -> batalla: ataqueMagico(intJugador)
        batalla -> personajeEnemigo: calcularDaño(daño)
        personajeEnemigo --> batalla: sigueVivo(bool)

        alt sigueVivo == true
            batalla -> usuario: terminarTurno()
        else sigueVivo == false
            batalla -> batalla: terminarBatalla()
            batalla -> usuario: notificarFinBatalla()
        end

    else manaDisponible == false
        personajeJugador --> usuario: mensajeMana()
        batalla -> usuario: continuarTurno()
        usuario -> usuario: seleccionarAccion()
    end

else accion = ataqueBasico
    usuario -> personajeJugador: ataqueBasico()
    personajeJugador -> batalla: ataqueFisico(strJugador)
    batalla -> personajeEnemigo: calcularDaño(daño)
    personajeEnemigo --> batalla: sigueVivo(bool)

    alt sigueVivo == true
        batalla -> usuario: terminarTurno()
    else sigueVivo == false
        batalla -> batalla: terminarBatalla()
        batalla -> usuario: notificarFinBatalla()
    end

else accion = defender
    usuario -> personajeJugador: defender()
    personajeJugador -> personajeJugador: aumentarDefensa()
    personajeJugador -> personajeJugador: recuperarMana()
    batalla -> usuario: terminarTurno()
end
